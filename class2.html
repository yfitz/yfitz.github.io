Deklaration von Funktionen

Um eine Funktion zu deklarieren schreibt man Code in{}

function parrotFacts() {
  console.log('Some parrot species can live for over 80 years');
  console.log('Kakapos are a critically endangered flightless parrot');
}

Argumente

Funktionen akzeptieren Eingabewerte, sogenannte Argumente.

function callKitten(kittenName){
  console.log('Come here, ' + kittenName + '!');
}

callKitten('Fluffy'); // outputs 'Come here, Fluffy!'

function addNumbers(a, b){
  console.log(a + b);
}

addNumbers(5, 7);  // outputs 12
addNumbers(9, 12); // outputs 21

Rückgabewerte

Werte, die beim Aufruf an die aufrufende Funktion zurückgegeben werden.

function square(num) {
  return num * num;
}

console.log(square(4));       // outputs '16'

var squareOfFive = square(5); // squareOfFive equals '25'

return beendet die Funktion augenblicklich.

Global Scope

Eine Variable, die ausserhalb von einer Funktion deklariert wurde:

var awesomeGroup = 'Girl Develop It'; // Global scope

function whatIsAwesome() {
  console.log(awesomeGroup + ' is pretty awesome.'); // Will work
}

whatIsAwesome();

Auf diese Variable kann von überall zugegriffen werden.

Local Scope

Eine Variable, die in einer Funktion deklariert wurde:

function whatIsAwesome() {
  var awesomeGroup = 'Girl Develop It';              // Local scope
  console.log(awesomeGroup + ' is pretty awesome.'); // Will work
}

whatIsAwesome();

console.log(awesomeGroup + ' is pretty awesome.'); // Won't work

Auf diese Variable kann nur innerhalb der Funktion zugegriffen werden.

Boolean Variablen

Boolean Variablen sind die Werte true and false

var catsAreBest = true;
var dogsRule = false;

Jeder Werte kann im boolean Kontext ausgewertet werden.

// Diese Variablen werden zu false ausgewertet
var myName = '';
var numOfKids = 0;
var isMarried;     // Variable ohne Werte ist undefined

null, NaN, undefined, 0, '' und "" wird zu false ausgewertet. Siehe: falsy
Alles andere wird zu true ausgewertet.

if Anweisung

var age = 30;

if (age > 18) {
  console.log('You are an adult');
}

if/else Anweisung

var age = 30;

if (age >= 16) {
  console.log('Yay, you can drive!');
} else {
  console.log('Sorry, you have ' + (16 - age) +
  ' years until you can drive.');
}

if/else statement

Mehrere Alternativen mit else if.

var age = 30;

if (age >= 35) {
  console.log('You can vote AND run for President!');
} else if (age >= 30) {
  console.log('You can vote AND run for the Senate!');
} else if (age >= 18) {
  console.log('You can vote!');
} else {
  console.log('You can\'t vote, but you can write your representatives.');
}

Vergleichs Operatoren

Beispiel Ergebnis
a == b TRUE wenn a gleich b (können unterschiedlichen Datentyp haben).
a=== b TRUE wenn a identisch mit b, und gleichen Datentyp.
a != b TRUE wenn a ungleich b können unterschiedlichen Datentyp haben).
a!== b TRUE wenn a nicht identisch zu b, oder nicht gleichen Datentyps.
a < b TRUE wenn a ist kleiner als b.
a > b TRUE wenn a ist grösser als b.
a <= b TRUE wenn a kleiner oder gleich b.
a >= b TRUE wenn a grösser oder gleich b.

Logische Operatoren

Beispiel Ergebnis
a && b TRUE wenn beide a und b TRUE sind.
a || b TRUE wenn entweder a oder b TRUE ist.
!a TRUE wenn a nicht TRUE ist.

Schleifen

while statement

while (x < 5) {
        console.log(x);
        x++;
}


for (initialize; condition; update) {
        // statements to repeat
}

Beispiel:

for (var i = 0; i < 5; i++) {
        console.log(i);
}

Break

Aus der Schleife aussteigen break statement.

// Count from 100 to 200
for (var i = 100; i <= 200; i++) {
  console.log('Testing ' + i);

  //Stop at the first multiple of 7
  if (i % 7 == 0) {
    console.log('Found it! ' + i);
    break;
  }
}
